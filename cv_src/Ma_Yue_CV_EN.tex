\section{Profile}

Experimental nuclear physicist and Senior Research Scientist (permanent) at RIKEN, with more than 15 years of front-line research and leadership at major international facilities in Japan (J-PARC, SPring-8, KEK) and Germany (GSI, DESY), and a long record of training young researchers and building international collaborations. \keypt{Currently experiment spokesperson of \tech{J-PARC E73} (hypertriton lifetime), and proposal spokesperson of \tech{SPring-8 LEPS2} (nucleon short-range correlations) and the \tech{J-PARC} antideuteron experiment.}

\section{Signature Achievements}
\begin{enumerate}
  \item \keypt{Discovery of the \Kmpp\ (\KNN) kaonic nuclear bound state} --- as a founding member, developed the multi-hit TDC and extracted the signal by polarization analysis, giving the first evidence for a deeply bound \Kmpp\ state (J-PARC E15). \,{\small Phys.\ Lett.\ B \textbf{789} (2019); Phys.\ Rev.\ C \textbf{102} (2020).}
  \item \keypt{Precise lifetime measurement of the \hn{4}{H} hypernucleus} --- as spokesperson and corresponding author, proposed the key idea of tagging the fast $\pi^0$ with a single high-energy $\gamma$ and led the measurement, $\tau=206\pm8\pm12$\,ps (J-PARC E73). \,{\small Phys.\ Lett.\ B \textbf{845} (2023); \textbf{873} (2026).}
  \item \keypt{First observation of a $\Lambda\pi$ structure near the \KN($I{=}1$) threshold} --- led the Belle data analysis as first author. \,{\small Phys.\ Rev.\ Lett.\ \textbf{130} (2023).}
  \item \keypt{First application of \tech{TES} microcalorimeters to hadronic-atom spectroscopy} --- implemented the SDD data analysis and energy calibration at sub-eV precision (J-PARC E62). \,{\small Prog.\ Theor.\ Exp.\ Phys.\ \textbf{2016}, 091D01; Phys.\ Rev.\ Lett.\ \textbf{128} (2022).}
  \item \keypt{First demonstration of energy-resolved muon spin spectroscopy ($\mu$SR)} --- as first and corresponding author, introduced event-by-event positron-energy selection with lead-glass Cherenkov calorimeters and led the demonstration, a 26\% figure-of-merit gain at no cost in beam time (J-PARC MLF). \,{\small Nucl.\ Instrum.\ Meth.\ A, in press (2026).}
\end{enumerate}
